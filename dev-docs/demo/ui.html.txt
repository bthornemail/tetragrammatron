// ── State ────────────────────────────────────────────────────────────────────
const state = {
  lastResult: null,
  lastSid: null,
  lastDescriptor: null,
  events: [],
  nrrLog: [],
  eventCount: 0,
}

// ── Node health ──────────────────────────────────────────────────────────────
async function checkNode() {
  try {
    const r = await fetch('/nrr/stats')
    if (r.ok) {
      const d = await r.json()
      document.getElementById('node-status').textContent = 'node running'
      document.getElementById('dot').className = 'dot'
      document.getElementById('sb-blobs').textContent = d.blobs
      document.getElementById('sb-entries').textContent = d.entries
      state.nrrLog = d.log ?? []
      if (document.getElementById('nrr-pane').style.display !== 'none') renderNRR()
    }
  } catch {
    document.getElementById('node-status').textContent = 'node offline'
    document.getElementById('dot').className = 'dot warn'
  }
}
setInterval(checkNode, 3000)
checkNode()

// ── SSE Event stream ──────────────────────────────────────────────────────────
function connectSSE() {
  const es = new EventSource('/events')
  es.onmessage = e => {
    const evt = JSON.parse(e.data)
    state.events.unshift(evt)
    state.eventCount++
    document.getElementById('sb-events').textContent = state.eventCount
    if (document.getElementById('events-pane').style.display !== 'none') renderEvents()
  }
  es.onerror = () => setTimeout(connectSSE, 2000)
}
connectSSE()

// ── Tab switching ─────────────────────────────────────────────────────────────
function setupTabs(barSelector, paneIds) {
  const tabs = document.querySelectorAll(barSelector + ' .tab')
  tabs.forEach(tab => {
    tab.addEventListener('click', () => {
      tabs.forEach(t => t.classList.remove('active'))
      tab.classList.add('active')
      paneIds.forEach(id => {
        document.getElementById(id).style.display =
          id === tab.dataset.tab + '-pane' ? 'flex' : 'none'
      })
      if (tab.dataset.tab === 'events') renderEvents()
      if (tab.dataset.tab === 'nrr')    { checkNode(); renderNRR() }
    })
  })
}
setupTabs('.panel:nth-child(2) .tab-bar', ['output-pane','graph-pane','descriptor-pane','raw-pane'])
setupTabs('.panel:nth-child(3) .tab-bar', ['events-pane','nrr-pane'])

// ── Resolve ───────────────────────────────────────────────────────────────────
document.getElementById('resolve-btn').addEventListener('click', async () => {
  const btn = document.getElementById('resolve-btn')
  btn.className = 'btn running'
  btn.textContent = '⟳ resolving…'

  let doc
  try { doc = JSON.parse(document.getElementById('doc-input').value) }
  catch (e) {
    showError('JSON parse error: ' + e.message)
    btn.className = 'btn'; btn.textContent = '↳ POST /resolve'; return
  }

  const stage = document.getElementById('stage-select').value
  const body = { canonical_input: { d: doc }, target_stage: stage }

  try {
    const r = await fetch('/resolve', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    })
    const data = await r.json()
    state.lastResult = data

    if (data.reject_kind) {
      showReject(data)
    } else {
      state.lastSid = data.sid
      state.lastDescriptor = data.descriptor
      showOutput(data)
      showGraph(data)
      showDescriptor(data)
      showRaw(data)
      if (data.sid) {
        document.getElementById('adapter-section').style.display = 'block'
        document.getElementById('adapter-sid-display').textContent = data.sid
      }
    }
    checkNode()
  } catch(e) {
    showError('fetch error: ' + e.message)
  }

  btn.className = 'btn'; btn.textContent = '↳ POST /resolve'
})

// ── Adapter ───────────────────────────────────────────────────────────────────

  if (!state.lastSid) return
  const label = document.getElementById('adapter-select').value
  const encodedLabel = encodeURIComponent(label)
  const encodedSid = encodeURIComponent(state.lastSid)
  try {
    const r = await fetch(`/adapter/${encodedLabel}/${encodedSid}`)
    const data = await r.json()
    document.getElementById('adapter-json').textContent = JSON.stringify(data, null, 2)
    document.getElementById('adapter-result').style.display = 'block'
  } catch(e) {
    document.getElementById('adapter-json').textContent = 'Error: ' + e.message
    document.getElementById('adapter-result').style.display = 'block'
  }
})

// ── Render: Graph ─────────────────────────────────────────────────────────────
function showGraph(data) {
  const pane = document.getElementById('graph-pane')
  if (!data.canonical_value?.edges) { pane.innerHTML = '<div class="empty-state"><div class="msg">No edges</div></div>'; return }

  const { nodes, edges } = data.canonical_value

  // Build canonical node set from edges after alias collapse
  const aliasMap = {}
  for (const [rel, a, b] of edges) {
    if (rel === 'alias' && a !== b) aliasMap[b] = a
  }

  const canonical = new Set()
  for (const n of nodes) {
    if (!aliasMap[n]) canonical.add(n)
  }

  let html = `<div class="section-label" style="margin-bottom:10px">Nodes</div>`
  html += `<div style="display:flex;flex-wrap:wrap;gap:6px;margin-bottom:16px">`
  for (const n of nodes) {
    const isCanon = canonical.has(n)
    html += `<span class="node-pill${isCanon ? ' canonical' : ''}">${n}</span>`
  }
  html += `</div>`

  html += `<div class="section-label" style="margin-bottom:10px">Edges</div>`
  for (const edge of edges) {
    const [rel, ...args] = edge
    if (rel === 'alias' && args[0] === args[1]) continue  // skip self-alias
    html += `<div class="graph-row">`
    html += `<span class="node-pill${canonical.has(args[0]) ? ' canonical' : ''}">${args[0]}</span>`
    html += `<span class="edge-pill"><span class="rel-name">${rel}</span></span>`
    html += `<span class="node-pill${canonical.has(args[1] ?? '') ? ' canonical' : ''}">${args[1] ?? ''}</span>`
    html += `</div>`
  }

  pane.innerHTML = html
  pane.style.display = 'flex'
  pane.style.flexDirection = 'column'
}

// ── Render: Events ────────────────────────────────────────────────────────────
function renderEvents() {
  const pane = document.getElementById('events-pane')
  if (!state.events.length) { pane.innerHTML = `<div style="color:var(--text3);font-size:10px;padding:8px 0">Waiting for events…</div>`; return }

  const evClassMap = {
    'resolve': 'ev-resolve',
    'identity': 'ev-identity',
    'descriptor': 'ev-descriptor',
    'nrr': 'ev-nrr',
    'node': 'ev-node',
    'adapter': 'ev-resolve',
  }

  let html = ''
  for (const e of state.events.slice(0, 60)) {
    const [family, ...rest] = e.event_kind.split('.')
    const cls = evClassMap[family] ?? ''
    const ts = new Date(e.ts ?? 0).toLocaleTimeString('en', { hour12: false, hour: '2-digit', minute: '2-digit', second: '2-digit' })
    html += `<div class="event-row ${cls}">
      <span class="event-ts">${ts}</span>
      <span class="event-kind">
        <span class="family">${family}.</span><span class="name">${rest.join('.')}</span>
      </span>
    </div>`
  }
  pane.innerHTML = html
}

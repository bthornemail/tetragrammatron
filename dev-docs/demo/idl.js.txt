/**
 * idl.js — DBC-IDL-1.2 identity layer (demo subset)
 *
 * Implements:
 *   - canonical TLV SID encoding (IDL §1.7)
 *   - SID derivation with domain tag (IDL §1.3)
 *   - sid:dbc: prefix namespace (IDL §1.8)
 *   - IdentityDescriptor projection (IDL §4)
 *
 * Laws demonstrated:
 *   SID Determinism       same inputs → same SID
 *   Canonical Encoding    same inputs → same SID_input_bytes
 *   Domain Tag            hash prefixed with "DBC-SID-1.1\0"
 *   Namespace             sid:dbc: prefix required in text form
 *   Descriptor Non-Authority  descriptor ↛ SID
 *   Meaning Primacy       identity is structural, not credential
 */

import { createHash } from 'crypto'
import { canonicalJSON } from './dbc.js'

// ── SID derivation ────────────────────────────────────────────────────────────

const DOMAIN_TAG = Buffer.from('DBC-SID-1.1\0', 'utf8')

/**
 * Canonical TLV encoding per IDL §1.7.
 * Fields in fixed order: schema_digest, normal_form, federation_scope, derivation_context
 */
function encodeSIDInput(schemaDigest, normalForm, federationScope, derivationContext) {
  function tlvField(name, value) {
    const nameBytes = Buffer.from(name, 'utf8')
    const nameLenBuf = Buffer.alloc(2)
    nameLenBuf.writeUInt16BE(nameBytes.length)

    const valueStr  = canonicalJSON(value ?? (typeof value === 'string' ? '' : {}))
    const valueBytes = Buffer.from(valueStr, 'utf8')
    const valueLenBuf = Buffer.alloc(4)
    valueLenBuf.writeUInt32BE(valueBytes.length)

    return Buffer.concat([nameLenBuf, nameBytes, valueLenBuf, valueBytes])
  }

  return Buffer.concat([
    tlvField('schema_digest',      schemaDigest),
    tlvField('normal_form',        normalForm),
    tlvField('federation_scope',   federationScope ?? ''),
    tlvField('derivation_context', derivationContext ?? {}),
  ])
}

/**
 * Derive a SID from a NormalForm.
 * Returns { sid, digest } where sid is "sid:dbc:<hex>".
 */
export function deriveSID(normalForm, schemaDigest, federationScope = null, derivationContext = {}) {
  if (!normalForm || normalForm.value_kind !== 'NormalForm') {
    throw new Error('IDL: SID may only be derived from a NormalForm')
  }

  const inputBytes = encodeSIDInput(
    schemaDigest,
    normalForm.value,
    federationScope,
    derivationContext
  )

  const hashInput = Buffer.concat([DOMAIN_TAG, inputBytes])
  const digest = createHash('sha256').update(hashInput).digest('hex')
  const sid = 'sid:dbc:' + digest

  return { sid, digest, schemaDigest, federationScope }
}

// ── IdentityDescriptor projection ─────────────────────────────────────────────

/**
 * Project an IdentityDescriptor from governed state.
 * Non-authoritative: this describes SID; it does not define it.
 * Per IDL §4.4 canonical JSON form.
 */
export function projectDescriptor(sidResult, normalForm, epoch = 'demo-epoch-1', services = [], governors = []) {
  const { sid, schemaDigest, federationScope } = sidResult

  const descriptor = {
    adapters:         {},
    epoch,
    federation_scope: federationScope ?? null,
    governors,
    revocation:       { endpoint: null, revocation_digest: null },
    schema_digest:    schemaDigest,
    services,
    sid,
    spec:             'DBC-IDL-1.2',
  }

  // descriptor_digest is sha256 of the canonical form with descriptor_digest = null
  const withNull = { ...descriptor, descriptor_digest: null }
  const descriptorDigest = 'sha256:' + createHash('sha256')
    .update(Buffer.from(canonicalJSON(withNull), 'utf8'))
    .digest('hex')

  descriptor.descriptor_digest = descriptorDigest

  return descriptor
}

// ── Adapter stubs (demo) ──────────────────────────────────────────────────────

/**
 * Demo adapter derivation. In the full stack this is HKDF per IDL §3.
 * For the demo, we derive a short deterministic value from the SID.
 */
export function deriveAdapter(sid, label) {
  const seed = createHash('sha256')
    .update(Buffer.from(sid + '\0' + label, 'utf8'))
    .digest('hex')

  switch (label) {
    case 'adapter:evm':
      return { adapter_label: label, credential: '0x' + seed.slice(0, 40), credential_kind: 'evm_address', sid }
    case 'adapter:did:key':
      return { adapter_label: label, credential: 'did:key:z6Mk' + seed.slice(0, 32), credential_kind: 'did_key', sid }
    case 'adapter:ipv6': {
      const b = Buffer.from(seed.slice(0, 32), 'hex')
      b[0] = 0xfd  // ULA prefix
      const groups = []
      for (let i = 0; i < 16; i += 2) {
        groups.push(b.slice(i, i + 2).toString('hex'))
      }
      return { adapter_label: label, credential: groups.join(':'), credential_kind: 'ipv6_address', sid }
    }
    default:
      return null
  }
}
